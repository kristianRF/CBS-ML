import seaborn as sns
import matplotlib.pyplot as plt

# Function to calculate correlation coefficient between two arrays
def corr(x, y, **kwargs):

    # Calculate the value
    coef = np.corrcoef(x, y)[0][1]
    # Make the label
    label = r"$\rho$ = " + str(round(coef, 2))

    # Add the label to the plot
    ax = plt.gca()
    ax.annotate(label, xy=(0.2, 0.95), size=10, xycoords=ax.transAxes)

def plot_roc_curve_compare(fpr_dict, tpr_dict):
    for i in fpr.keys():
        plt.plot(
            fpr_dict[i],
            tpr_dict[i],
            label="%s: AUC = %0.4f" % (i, auc(fpr_dict[i], tpr_dict[i])),
        )
    plt.plot([0, 1], [0, 1], color="Green", linestyle="--", label="Random: AUC=0.50")
    plt.xlabel("False positive rate")
    plt.ylabel("True positive rate")
    plt.axis("square")
    plt.title("Receiver Operating Characteristic (ROC) Curve")
    plt.legend(loc="lower right")
    plt.show()
